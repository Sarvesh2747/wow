def is_safe(node, graph, color, c):

    for neighbor in graph[node]:

        if color[neighbor] == c:
            return False

    return True


def graph_coloring(graph, m, color, node, n):

    # Base condition
    if node == n:

        print("Solution:", color)
        return True

    # Try different colors
    for c in range(1, m + 1):

        if is_safe(node, graph, color, c):

            color[node] = c

            graph_coloring(graph, m, color, node + 1, n)

            # Backtracking
            color[node] = 0

    return False


n = int(input("Enter number of vertices: "))

graph = {}

for i in range(n):

    graph[i] = []

    e = int(input(f"Enter neighbors count for vertex {i}: "))

    for j in range(e):

        neighbor = int(input("Enter neighbor: "))

        graph[i].append(neighbor)


m = int(input("Enter number of colors: "))

color = [0] * n

graph_coloring(graph, m, color, 0, n)
