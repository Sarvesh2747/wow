class Node:

    def __init__(self, position, g, h, parent):

        self.position = position
        self.g = g
        self.h = h
        self.f = g + h
        self.parent = parent


def heuristic(a, b):

    # Manhattan Distance
    return abs(a[0] - b[0]) + abs(a[1] - b[1])


def astar(grid, start, goal):

    open_list = []
    closed_set = set()

    start_node = Node(start, 0, heuristic(start, goal), None)

    open_list.append(start_node)

    while open_list:

        # Find node with minimum f
        current = min(open_list, key=lambda x: x.f)

        open_list.remove(current)

        # Goal check
        if current.position == goal:

            path = []

            while current:
                path.append(current.position)
                current = current.parent

            return path[::-1]

        closed_set.add(current.position)

        x, y = current.position

        neighbors = [
            (x+1, y),
            (x-1, y),
            (x, y+1),
            (x, y-1)
        ]

        for pos in neighbors:

            # Boundary check
            if pos[0] < 0 or pos[1] < 0 or pos[0] >= len(grid) or pos[1] >= len(grid[0]):
                continue

            # Obstacle check
            if grid[pos[0]][pos[1]] == 1:
                continue

            # Closed check
            if pos in closed_set:
                continue

            g = current.g + 1
            h = heuristic(pos, goal)

            neighbor_node = Node(pos, g, h, current)

            open_list.append(neighbor_node)

    return None


rows = int(input("Enter rows: "))
cols = int(input("Enter columns: "))

grid = []

print("Enter grid values:")

for i in range(rows):

    row = list(map(int, input().split()))
    grid.append(row)

start = tuple(map(int, input("Enter start position: ").split()))
goal = tuple(map(int, input("Enter goal position: ").split()))

path = astar(grid, start, goal)

print("Shortest Path:", path)
